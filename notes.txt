CREATE TABLE users_csv (
  id INT,
  name VARCHAR(100),
  email VARCHAR(255)
)
ENGINE=CONNECT
TABLE_TYPE=CSV
FILE_NAME='/var/lib/mysql/users.csv'
HEADER=1;

CREATE TABLE users_json (
  name VARCHAR(100),
  age INT
)
ENGINE=CONNECT
TABLE_TYPE=JSON
FILE_NAME='/var/lib/mysql/users.json'

CREATE TABLE users_xml (
  id INT,
  name VARCHAR(100),
  age INT
)
ENGINE=CONNECT
TABLE_TYPE=XML
FILE_NAME='/var/lib/mysql/users.xml'
ROW_TAG='person';
